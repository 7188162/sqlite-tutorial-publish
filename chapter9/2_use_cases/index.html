SELECT
  c.Prefecture
FROM
  Sales AS s
JOIN
  Customers AS c ON s.CustomerID = c.CustomerID
WHERE
  strftime('%Y', s.SaleDate) = '2024'
GROUP BY
  c.Prefecture
ORDER BY
  SUM(s.Amount) DESC;
